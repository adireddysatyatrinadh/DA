SELECT UPPER(JOB_CD) FROM  ETL_JOB_WORKER_JOBS

SELECT UPPER(JOB_WORKER_CD) FROM ETL_JOB_WORKER
WHERE UPPER(JOB_AGENT_CD) LIKE '%CDB2FILE%'
));

WHERE  UPPER(JOB_CD) IN (
SELECT UPPER(JOB_CD) FROM  ETL_JOB_WORKER_JOBS
WHERE UPPER(JOB_WORKER_CD) IN (
SELECT UPPER(JOB_WORKER_CD) FROM ETL_JOB_WORKER
WHERE UPPER(JOB_AGENT_CD) LIKE '%CDB2FILE%'
));
